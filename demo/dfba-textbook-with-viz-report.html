Composite spec (YAML)

name: dfba-textbook-with-viz
description: "Dynamic FBA demo with a wired Visualization Step. Each update produces a Plotly HTML figure of the running scalar + per-species trajectories, written to the 'viz_html' store."

requires:
  processes: [DynamicFBAProcess, DynamicFBAPlots, RAMEmitter]

parameters:
  interval:
    type: float
    default: 1.0
    description: "Process interval in hours."
  substep:
    type: float
    default: 0.2
    description: "Internal Euler integration step (hr)."
  initial_biomass:
    type: float
    default: 0.001
    description: "Initial E. coli biomass (gDW)."
  initial_glucose:
    type: float
    default: 10.0
    description: "Initial glucose in the media (mmol)."
  viz_title:
    type: string
    default: "COMETS dynamic FBA"
    description: "Title displayed at the top of the Plotly figure."

state:
  community:
    _type: process
    address: "local:DynamicFBAProcess"
    config:
      models: ['textbook']
      model_ids: ['Ecoli']
      initial_biomass: ["${initial_biomass}"]
      initial_media:
        glc__D: "${initial_glucose}"
        o2: 40.0
        nh4: 1000.0
        pi: 1000.0
        h2o: 1000.0
        co2: 1000.0
        h: 1000.0
        ac: 0.0
      substep: "${substep}"
    inputs: {}
    outputs:
      biomass: ["stores", "biomass"]
      media: ["stores", "media"]
      growth_rates: ["stores", "growth_rates"]
      total_biomass: ["stores", "total_biomass"]
    interval: "${interval}"
  stores:
    biomass: {}
    media: {}
    growth_rates: {}
    total_biomass: 0.0
    viz_html: ""
  viz:
    _type: step
    address: "local:DynamicFBAPlots"
    config:
      title: "${viz_title}"
    inputs:
      total_biomass: ["stores", "total_biomass"]
      biomass: ["stores", "biomass"]
      growth_rates: ["stores", "growth_rates"]
      time: ["global_time"]
    outputs:
      html: ["stores", "viz_html"]
  emitter:
    _type: step
    address: "local:ram-emitter"
    config:
      emit:
        biomass: "map[float]"
        media: "map[float]"
        growth_rates: "map[float]"
        total_biomass: "float"
        time: "float"
    inputs:
      biomass: ["stores", "biomass"]
      media: ["stores", "media"]
      growth_rates: ["stores", "growth_rates"]
      total_biomass: ["stores", "total_biomass"]
      time: ["global_time"]
